import javax.swing.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
public class fibonacci {
    private JTextField valorFibonacci;
    private JTextField result;
    private JButton consultarButton;
    private JButton limpiarButton;
    private JButton salirDelProgramaButton;
    private JPanel nico;
    private JTextField Factorial;
    private JLabel Calcular;
    private JButton factorial;
    private JTextField valorf;
    private JTextField valorM;
    private JButton multiplo;
    private JTextField valorm2;
    private JTextArea Result;

    public fibonacci() {
        consultarButton.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                int n;
                n = Integer.parseInt(valorFibonacci.getText());
                int a = 0, b = 1, c = 0, s = 1;
                for (int i = 2; i <= n; i++) {
                    c = a + b;
                    s += c;
                    a = b;
                    b = c;
                }
                result.setText(Integer.toString(c));
            }
        });
        limpiarButton.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                valorFibonacci.setText("");
                valorf.setText("");
                valorM.setText("");
                valorm2.setText("");
                result.setText("");
                Result.setText("");
            }
        });
        salirDelProgramaButton.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {

                System.exit(0);
            }
        });

        factorial.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                try{
                    int factoriall = 1;
                    Integer n = new Integer(valorf.getText());

                    for(int i = n; i >= 1; i--){
                        factoriall = factoriall * i;
                    }
                    result.setText(" " + factoriall);
                }
                catch(NumberFormatException c){
                    result.setText("Solo puede ingresar n√∫meros");
                }
            }
        });

        multiplo.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
            int m = Integer.parseInt(valorM.getText());
            int m1 = Integer.parseInt(valorm2.getText());
            String resultado = "";
            for (int i = 0; i < m1; i++ ) {
                if (i % m == 0) {
                    resultado = resultado + i + "\n";
                }
            }
            Result.append(String.valueOf("Resultado:\n "  + resultado));
            }
        });
    }

    public static void main(String[]args){
        JFrame frame = new JFrame("Programa");
        frame.setContentPane(new fibonacci().nico);

        frame.pack();
        frame.setVisible(true);
    }
}
